\subsection{API Contract}\label{sec:api}
All endpoints must follow RESTful conventions. RESTful endpoints with JSON request/response bodies and expected HTTP status codes.
Each exchange shows a raw HTTP request or response trace: status line, headers, a blank line, then the body.
All paths are relative to the API base URL.

\medskip
\noindent\textbf{DELETE responses.} A successful delete returns \texttt{HTTP/1.1 204 No Content} with no
response body---the server has fulfilled the request and there is no representation to return (RFC~9110).
Error cases (e.g.\ resource not found) return \texttt{4xx} with a JSON body as documented below.

\subsubsection{Endpoint Summary}
\begin{tabularx}{\textwidth}{@{} l l X @{}}
  \toprule
  \textbf{Method} & \textbf{Path} & \textbf{Purpose} \\
  \midrule
  \texttt{POST}   & \texttt{/api/auth/register}           & Register a new account \\
  \texttt{POST}   & \texttt{/api/auth/login}              & Authenticate an existing account \\
  \texttt{POST}   & \texttt{/api/todos}                   & Create a todo \\
  \texttt{GET}    & \texttt{/api/todos}                   & List todos for the authenticated user \\
  \texttt{GET}    & \texttt{/api/todos/\{id\}}            & Retrieve a single todo \\
  \texttt{PUT}    & \texttt{/api/todos/\{id\}}            & Update a todo \\
  \texttt{DELETE} & \texttt{/api/todos/\{id\}}            & Delete a todo (cascade subtasks) \\
  \texttt{GET}    & \texttt{/api/todos/\{id\}}            & Retrieve parent todo (before subtask operations) \\
  \texttt{POST}   & \texttt{/api/todos/\{id\}/\{subtask\}} & Create a subtask \\
  \texttt{PUT}    & \texttt{/api/todos/\{id\}/\{subtask\}} & Update a subtask \\
  \texttt{DELETE} & \texttt{/api/todos/\{id\}/\{subtask\}} & Delete a subtask \\
  \bottomrule
\end{tabularx}

\subsubsection{Account Creation}
Use the resource-creation flow template for this \texttt{POST} endpoint (Figure~\ref{fig:seq-register}).

\textbf{\texttt{POST /api/auth/register}}

\begin{httpexchange}
\begin{httplisting}
POST /api/auth/register HTTP/1.1
Content-Type: application/json

{
    "username": "String",
    "password": "String"
}
\end{httplisting}
\httpmidarrows
\begin{httplisting}
HTTP/1.1 200 OK
Content-Type: application/json

{
    "accountId": Integer,
    "username": "String"
}

HTTP/1.1 409 Conflict
Content-Type: application/json

{
    "status": 409
}

HTTP/1.1 400 Bad Request
Content-Type: application/json

{
    "status": 400
}
\end{httplisting}
\end{httpexchange}

\textbf{\texttt{DELETE /api/todos/\{id\}/\{subtask\}}}

\texttt{204 No Content} and an empty body; no row count or deleted entity is returned.

\begin{httpexchange}
\begin{httplisting}
DELETE /api/todos/{id}/{subtask} HTTP/1.1
\end{httplisting}
\httpmidarrows
\begin{httplisting}
HTTP/1.1 204 No Content

HTTP/1.1 404 Not Found
Content-Type: application/json

{
    "status": 404
}
\end{httplisting}
\end{httpexchange}
